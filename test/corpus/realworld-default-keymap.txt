DefaultKeyMap(defaults*) {
    return {
        Call: (t, v*) => dkm(v*),
        __Enum: (t, n) => defaults.__Enum(n)
    }

    dkm(values*) {
        interleaved := []
        e1 := defaults.__Enum(1), e2 := values.__Enum(1)
        while e1(&k) && e2(&v)
            interleaved.Push(k, v)
        return Map(interleaved*)
    }
}


DKM1 := DefaultKeyMap("k1", "k2")
myMap := DKM1("v1", "v2")

for key in DKM1
    MsgBox "DKM1 " A_Index ": " key

for key, value in myMap
    MsgBox "myMap " key " -> " value
